<script>
  const productTypeSelect = document.getElementById('productType');
  const sizeSelect = document.getElementById('sizeSelect');
  const containerSelect = document.getElementById('containerSelect');
  const waxTypeSelect = document.getElementById('waxTypeSelect');
  const scentSelect = document.getElementById('scentSelect');
  const colorSelect = document.getElementById('colorSelect');
  const siliconeMoldSection = document.getElementById('siliconeMoldSection');

  // Option data based on your specs
  const optionsData = {
    waxMelts: {
      sizes: ['2.5 oz'],
      containers: ['None'],
      waxTypes: ['Paraffin', 'Soy'],
      scents: ['Floral', 'Fresh', 'Warm/Sweet'],
      colors: ['Natural', 'White', 'Cool Gray', 'Light Blue', 'Midnight Blue', 'Moss Green', 'Terracotta', 'Burnt Orange', 'Plum', 'Crimson Red', 'Soft Yellow']
    },
    waxBars: {
      sizes: ['1.5 oz'],
      containers: ['None'],
      waxTypes: ['Paraffin', 'Soy'],
      scents: ['Floral', 'Fresh', 'Warm/Sweet'],
      colors: ['Natural', 'White', 'Cool Gray', 'Light Blue', 'Midnight Blue', 'Moss Green', 'Terracotta', 'Burnt Orange', 'Plum', 'Crimson Red', 'Soft Yellow']
    },
    waxCubes: {
      sizes: ['1 oz'],
      containers: ['None'],
      waxTypes: ['Paraffin', 'Soy'],
      scents: ['Floral', 'Fresh', 'Warm/Sweet'],
      colors: ['Natural', 'White', 'Cool Gray', 'Light Blue', 'Midnight Blue', 'Moss Green', 'Terracotta', 'Burnt Orange', 'Plum', 'Crimson Red', 'Soft Yellow']
    },
    candle: {
      sizes: ['4 oz', '6 oz', '8 oz', '15 oz'],
      containers: ['Glass Jar', 'Tin Jar'],
      waxTypes: ['Apricot Coconut Wax Blend', 'Soy Wax', 'Beeswax', 'Coconut Wax', 'Coconut & Beeswax Blend', 'Coconut Paraffin Wax Blend', 'Paraffin Blend', 'Palm Paraffin Blend', 'Palm Wax'],
      scents: ['Floral', 'Fresh', 'Warm/Sweet'],
      colors: ['Natural', 'White', 'Cool Gray', 'Light Blue', 'Midnight Blue', 'Moss Green', 'Terracotta', 'Burnt Orange', 'Plum', 'Crimson Red', 'Soft Yellow']
    },
    siliconeMoldedCandles: {
      sizes: [],
      containers: ['Silicone Mold'],
      waxTypes: ['Paraffin', 'Soy'],
      scents: ['Floral', 'Fresh', 'Warm/Sweet'],
      colors: ['Natural', 'White', 'Cool Gray', 'Light Blue', 'Midnight Blue', 'Moss Green', 'Terracotta', 'Burnt Orange', 'Plum', 'Crimson Red', 'Soft Yellow']
    }
  };

  function populateSelect(selectElem, options) {
    selectElem.innerHTML = ''; // Clear previous options
    if (options.length === 0) {
      selectElem.disabled = true;
      selectElem.innerHTML = '<option value="">N/A</option>';
      return;
    }
    selectElem.disabled = false;
    options.forEach(opt => {
      const option = document.createElement('option');
      option.value = opt;
      option.textContent = opt;
      selectElem.appendChild(option);
    });
  }

  productTypeSelect.addEventListener('change', () => {
    const selected = productTypeSelect.value;

    // Populate sizes
    populateSelect(sizeSelect, optionsData[selected].sizes);

    // Populate containers
    populateSelect(containerSelect, optionsData[selected].containers);

    // Populate wax types
    populateSelect(waxTypeSelect, optionsData[selected].waxTypes);

    // Populate scents
    populateSelect(scentSelect, optionsData[selected].scents);

    // Populate colors
    populateSelect(colorSelect, optionsData[selected].colors);

    // Show silicone mold checkboxes only for silicone molded candles
    if (selected === 'siliconeMoldedCandles') {
      siliconeMoldSection.classList.remove('hidden');
      containerSelect.disabled = true; // Since container is always silicone mold here
    } else {
      siliconeMoldSection.classList.add('hidden');
      containerSelect.disabled = false;
    }
  });
</script>
